Асинхронность JS

  • Никуда от нее не деться (UI должен жить)
  • Камень преткновения, особенно, если вы пришли из чего-то синхронного вроде PHP
  • Боль вложенности (callbacks, promises)

Как хочется чтобы все работало?


        function getUserResume(userId) {
          var user = fetchUserData(userId);
          var resume = fetchResume(user.resumeId);

          return resume;
        }

Callbacks спешат на помощь 


function getUserResume(userId, callback) {
  fetchUserData(userId, function(error, user) {
    if (error) {
      callback(error);
    } else {
      fetchResume(user.resumeId, function(error, resume) {
        if (error) {
          callback(error);
        } else {
          callback(resume);
        }
      })
    }
  });
}

магия генераторов


        function* getUserResume(userId) {
          let user = yield fetchUserData(userId);
          let resume = yield fetchResume(user.resumeId);

          return resume;
        }
            

        spawn(getUserResume('msxMqjP2bzBSdi5sg8zhFNxduK7FZvteEG'))

Однако, чтобы продолжить, нам нужно разобраться в том, что же такое генераторы

Generators

Двусторонняя коммуникация yield


  function *sumGenerator() {
      let x = yield "First value";
      let y = yield "Second value";
      return x + y;
  }
          

  let sum = sumGenerator();
  sum.next();    // {"value":"First value","done":false}
  sum.next(3);   // {"value":"Second value","done":false}
  sum.next(8);   // {"value":11,"done":true}
          

  function* getUserResume(userId) {
    let user = yield fetchUserData(userId);         // Promise
    let resume = yield fetchResume(user.resumeId);  // Promise

    return resume;
  }

  spawn(getUserResume('msxMqjP2bzBSdi5sg8zhFNxduK7FZvteEG'))
      .then(
          (resume) => console.log(resume),
          (error) => console.error(error));
          

  function spawn (generator) {
    return new Promise((resolve, reject) => {
      let onResult = lastPromiseResult => {
        let {value, done} = generator.next(lastPromiseResult);
        if (!done) {
          value.then(onResult, reject);
        }
        else resolve(value);
      }
      onResult();
    })
  }

В ES2016 анонсирована Async

  • Появилось две новых команды: async и await
  • В сухом остатке это новый дизайн для все тех же Promise

            async function foo() {
              if( Math.round(Math.random()) )
                return 'Success!';
              else
                throw 'Failure!';
            }
          

            function foo() {
              if( Math.round(Math.random()) )
                return Promise.resolve('Success!');
              else
                return Promise.reject('Failure!');
            }

Таким образом все ту же функцию getUserResume() мы можем переписать с использованием нового асинхронного функционала 


      function* getUserResume(userId) {
        let user = yield fetchUserData(userId);         // Promise
        let resume = yield fetchResume(user.resumeId);  // Promise

        return resume;
      }
          

      async function getUserResume(userId) {
        let user = await fetchUserData(userId);         // Promise
        let resume = await fetchResume(user.resumeId);  // Promise

        return resume;
      }

Это означает, что в этом случае нам нет необходимости использовать генераторы, а можно обойтись только Promise. 


      async function getUserResume(userId) {
        let user = await fetchUserData(userId);         // Promise
        let resume = await fetchResume(user.resumeId);  // Promise

        return resume;
      }

      // Вспомогательная функция spawn() больше не требуется

      getUserResume('msxMqjP2bzBSdi5sg8zhFNxduK7FZvteEG')
        .then(
          (resume) => console.log(resume),
          (error) => console.error(error));
